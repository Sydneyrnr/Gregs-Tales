        function startGame()
        {
                var gender = "boy";
                var city = " ";
                var monster = " ";
                var food = " ";
                var drink = " ";
                numYears = prompt("Enter a number greater than 0: ");
                gender = prompt("Is the story about a boy or a girl? ");
                letter = gender.charAt(0);
                gender = (letter == "b" || letter == "B") ? "boy" : "girl";
                pronoun = (gender == "boy") ? "he" : "she";
                name = prompt("Enter the hero's name: ");
                city = prompt("Enter the name of a city: ");
                monster = prompt("Enter a type of monster: ");
                food = prompt("Enter a food you like: ");
                drink = prompt("Enter a drink you like: ");
                document.getElementById("content").innerHTML = ("Once upon a time, about " + numYears 
                        + " years ago, there was a " + gender + " named " + name + ". " + name 
                        + " lived in a small cabin in the woods just outside " 
                        + city + " limits.</p> <p>" 
                        + name + " enjoyed walking in the woods every day until... One day " 
                        + pronoun + " came upon a " + monster + " sitting on a log eating a " 
                        + food + ".</p> <p>The " + monster + " jumped up, spilling his " 
                        + drink + ". " + name + " ran home as fast as " + pronoun + " could but the " 
                        + monster + " followed and ...</p><h3>What happened? You decide! </h3> <p>Ending 1: The " 
                        + monster + " and " + name + " became best friends and lived in " + name + "'s house happily ever after.</p> <p>Ending 2: The " 
                        + monster + " overpowered " + name + " and gobbled down all the " + food + " and " + drink + " in " + name + "'s refrigerator.</p> <p>Ending 3: " 
                        + name + " screamed mean things at the " + monster + ", causing the " 
                        + monster + " to turn and run back to the woods, never to be seen again.</p>");
            }
            function game2()
            {
            var num = 0;
            var properNoun = "";
            var verb = "";
            var pronoun = "";
            var adjective = "";

            num = prompt("Enter a number: ");
            properNoun = prompt("Enter a proper noun (a name): ");
            verb = prompt("Enter a verb: ");
            pronoun = prompt("Enter a pronoun: ");
            adjective = prompt("Enter an adjective: ");

            document.getElementById("content1").innerHTML = ("<p>Once upon a time, there was a courageous traveler named " 
                    + properNoun + ". " +"With " 
                    + pronoun + " trusty " 
                    + adjective + " stick, " 
                    + properNoun + " set out on a journey to find treasure.</p>" + "<p>After traveling for " + num + " days, " 
                    + properNoun + " reached a dark tunnel. Inside the tunnel, " 
                    + pronoun + " " + verb +" through dark passages and ran into many dangers.</p>" + "<p>Finally, "
                    + properNoun +" arrived at the heart of the cave, " 
                    + pronoun + " found a hidden chest filled with gold blocks and tons of jewelry. " + "With " 
                    + pronoun + " newfound riches, " + properNoun + " returned home triumphant.</p>");
            }
